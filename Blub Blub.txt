class Main {
	ListenOnTick = List();
	ListenOnFrame = List();
	ListenOnSecond = List();
	ListenOnChatInput = List();
	ListenOnCharacterSpawn = List();
	ListenOnCharacterDie = List();
	ListenOnCharacterDamaged = List();
	ListenOnNetworkMessage = List();
	ListenOnAction = List();
	ListenOnButtonClick = List();
	ListenOnLoadMap = List();
	ListenOnFinishRace = List();

	_modes = Dict();
	_currentMode = null;
	_nextMode = null;

	function Init() {
		MainController.InitMain(self);
	}

	function OnGameStart() {
		NetworkUtil.SetCustomProperty("bombround", 1);
		UI.SetScoreboardProperty("ScoreStatus");

		standardMap = MapSubset(Map.FindAllMapObjects());
		asoMap = standardMap.CreateCopy(Vector3(100000, 0, 0));
		bombMap = standardMap.CreateCopy(Vector3(0, 0, 100000));
		hubMap = MapLibrary.CreateHubMap(Vector3(-100000, 0, 0));
		akinaMap = Vector3(0, 0, -100000);

		if (Network.IsMasterClient) {
			MC_MapTransferLogic().Activate(self);
			MC_StandardLogic(standardMap).Activate(self);
			MC_BombLogic(bombMap).Activate(self);
			MC_AsoLogic(asoMap).Activate(self);
			MC_LeaderboardLogic().Activate(self);
		}

		self.UX = UILogic();
		self.AFK = AfkLogic();
		self.MapLoader = MapLoaderLogic();

		self.UX.Activate(self);
		self.AFK.Activate(self);
		self.MapLoader.Activate(self);
		LeaderboardLogic().Activate(self);

		self._modes.Set("WAVE", StandardLogic(standardMap));
		self._modes.Set("ASO", AsoLogic(asoMap));
		self._modes.Set("BOMB", BombLogic(bombMap));
		self._modes.Set("HUB", HubLogic(hubMap));
		self._modes.Set("AKINA", AkinaLogic(akinaMap));

		NetworkUtil.SetCustomProperty("CurrentMode", "");
		self.SwitchMode("WAVE");
		self.UpdateMode();
	}

	function SwitchMode(mode) {
		if (self._modes.Get(mode).CanJoin()) {
			self._nextMode = mode;
			return true;
		} else {
			return false;
		}
	}

	function UpdateMode() {
		if (self._nextMode == null) {
			return;
		}

		currentMode = Network.MyPlayer.GetCustomProperty("CurrentMode");

		if (currentMode != self._nextMode) {
			if (self._currentMode != null) {
				self._currentMode.Deactivate(self);
			}

			NetworkUtil.SetCustomProperty("CurrentMode", self._nextMode);
			self._currentMode = self._modes.Get(self._nextMode);
			self._currentMode.Activate(self);
			Network.MyPlayer.ClearKDR();
		}

		self._nextMode = null;
	}

	function OnChatInput(message) {
		for (logic in self.ListenOnChatInput) {
				if (logic.OnChatInput(message) == false) { return false; }
		}
		return true;
	}

	function FinishRace() {
			for (logic in self.ListenOnFinishRace) { logic.OnFinishRace(); }
	}

	function OnAction(action) {
			for (logic in self.ListenOnAction) { logic.OnAction(action); }
	}

	function OnLoadMap(map) {
			for (logic in self.ListenOnLoadMap) { logic.OnLoadMap(map); }
	}

	function OnButtonClick(button) {
			for (logic in self.ListenOnButtonClick) { logic.OnButtonClick(button); }
	}

	function OnTick() {
			for (logic in self.ListenOnTick) { logic.OnTick(); }
	}

	function OnSecond() {
			for (logic in self.ListenOnSecond) { logic.OnSecond(); }
		self.UpdateMode();
	}

	function OnCharacterSpawn(character) {
			for (logic in self.ListenOnCharacterSpawn) { logic.OnCharacterSpawn(character); }
	}

	function OnCharacterDie(victim, killer, killerName) {
			for (logic in self.ListenOnCharacterDie) { logic.OnCharacterDie(victim, killer, killerName); }
	}

	function OnCharacterDamaged(victim, killer, killerName, damage) {
			for (logic in self.ListenOnCharacterDamaged) { logic.OnCharacterDamaged(victim, killer, killerName, damage); }
	}

	function OnNetworkMessage(sender, message) {
			for (logic in self.ListenOnNetworkMessage) { logic.OnNetworkMessage(sender, message); }
	}

	function OnFrame() {
			for (logic in self.ListenOnFrame) { logic.OnFrame(); }
	}
}

class UILogic {
	_keybinds = Dict();

	function Activate(main) {
		main.ListenOnFrame.Add(self);
		main.ListenOnChatInput.Add(self);
		main.ListenOnNetworkMessage.Add(self);
		self.CreateJoinPopup();
		self.CreateGenericPopup();
	}

	function Deactivate(main) {
		main.ListenOnFrame.Remove(self);
		main.ListenOnChatInput.Remove(self);
		main.ListenOnNetworkMessage.Remove(self);
	}

	function CreateGenericPopup() {
		UI.CreatePopup("GenericPopup", "", 800, 600);
	}

	function CreateJoinPopup() {
		text = "";
		text += "<color=#00ccff>Blub Blub</color>" + String.Newline;
		text += "   ◦    " + String.Newline;
		text += "        ◦   " + String.Newline;
		text += " ◦    " + String.Newline;
		text += "      ◦   " + String.Newline;
		text += " ◦    " + String.Newline;
		text += "      ◦      " + String.Newline;
		text += "                       <°)))><◦°" + String.Newline;
		text += String.Newline;
		text += "- Commands -" + String.Newline;
		text += "!hub  |  !wave  |  !aso  |  !bomb  |  !akina" + String.Newline;
		text += "!leaderboard" + String.Newline;
		UI.CreatePopup("WelcomeMenu", "Welcome", 500, 600);
		UI.AddPopupLabel("WelcomeMenu", text);
	# UI.ShowPopup("WelcomeMenu");
	}

	function AddBind(key, action) {
		self._keybinds.Set(key, action);
		self.UpdateBindLabel();
	}

	function RemoveBind(key) {
		self._keybinds.Remove(key);
		self.UpdateBindLabel();
	}

	function OnNetworkMessage(sender, message) {
		dict = Json.LoadFromString(message);

		if (dict.Get("op") == "UI_popup") {
			UI.ClearPopup("GenericPopup");
			UI.AddPopupLabel("GenericPopup", dict.Get("body"));
			UI.ShowPopup("GenericPopup");
		}
	}

	function OnChatInput(message) {
			if (message == "!wave") { return MainController.SwitchMode("WAVE"); }
			elif (message == "!aso") { return MainController.SwitchMode("ASO"); }
			elif (message == "!bomb") { return MainController.SwitchMode("BOMB"); }
			elif (message == "!hub") { return MainController.SwitchMode("HUB"); }
			elif (message == "!akina") { return MainController.SwitchMode("AKINA"); }
			elif (message == "!leaderboard") { return MainController.TriggerAction("LeaderboardShow"); }
			elif (message == "!help") { UI.ShowPopup("WelcomeMenu"); }
		elif (String.StartsWith(message, "!roll")) {
			self.SayAfter("<color=white>" + Network.MyPlayer.Name + "</color> rolled for <color=green>" + Random.RandomInt(0, 101) + "</color>.");
		}
	}

	coroutine SayAfter(message) {
		wait 0.5;
		Game.PrintAll(message);
	}

	function UpdateBindLabel() {
		label = "<color=#B7E0FF>- Bubble -</color>";
		for (key in self._keybinds.Keys) {
			label += String.Newline + self._keybinds.Get(key) + " <color=#ffcc00>[" + Input.GetKeyName(key) + "]</color> ";
		}
		UI.SetLabel("BottomRight", label);
	}

	function OnKeyDown(action) {
		MainController.TriggerAction(action);
	}

	function OnFrame() {
		for (key in self._keybinds.Keys) {
			if (Input.GetKeyDown(key)) {
				self.OnKeyDown(self._keybinds.Get(key));
			}
		}
	}
}

class BombLogic {

	_lastBombRound = 0;
	function Init(map) {
		self._map = map;
		self._playerSpawner = PlayerSpawner(map);
	}

	function Activate(main) {
		main.ListenOnNetworkMessage.Add(self);
		main.ListenOnCharacterSpawn.Add(self);
		main.ListenOnTick.Add(self);
		currentBombRound = Network.MasterClient.GetCustomProperty("bombround");
		if (self._lastBombRound != currentBombRound){
			self._playerSpawner.Spawn();
			self._lastBombRound = currentBombRound;
		}
		else{
			me = Network.MyPlayer.Character;
			if (me != null){
				me.GetKilled("");
			}
		}
	}

	function Deactivate(main) {
		currentBombRound = Network.MasterClient.GetCustomProperty("bombround");
		self._lastBombRound = currentBombRound;
		main.ListenOnNetworkMessage.Remove(self);
		main.ListenOnCharacterSpawn.Remove(self);
		main.ListenOnTick.Remove(self);
	}

	function OnNetworkMessage(sender, message) {
		if (sender != Network.MasterClient) {
			return;
		}

		data = Json.LoadFromString(message);
		if (data.Get("op") == "BOMB_end") {
			self.OnEnd(data.Get("reason"));
		} elif (data.Get("op") == "BOMB_start") {
			self._playerSpawner.Spawn();
		}
	}

	function OnEnd(reason) {
		if (Network.MyPlayer.Status == "Alive") {
			self.KillAfter(1);
		}

		UI.SetLabelForTime("MiddleCenter", "<size=100>" + String.Newline + "</size>" + reason, 3);
	}

	coroutine KillAfter(time) {
		wait time;
		me = Network.MyPlayer.Character;
		if (me != null) {
			me.GetKilled("");
			Network.MyPlayer.Deaths -= 1;
		}
	}

	function OnTick() {
		me = Network.MyPlayer.Character;
		if (me != null) {
			me.CurrentGas = me.MaxGas;
		}

		UI.SetLabel("TopCenter", Network.MasterClient.GetCustomProperty("BOMB_status"));

		status = "[BOMB] " + Network.MyPlayer.Kills + "/" + Network.MyPlayer.Deaths;
		NetworkUtil.SetCustomProperty("ScoreStatus", status);
	}

	function OnCharacterSpawn(character) {
		if (character.IsMine && character.Type == "Human") {
			character.SetWeapon("Thunderspear");
			character.Team = "None";
		}
	}

		function CanJoin() { return true; }
}

class MC_BombLogic {
	_end = false;
	_bombRound = 1;

	function Init(map) { }

	function Activate(main) {
		main.ListenOnSecond.Add(self);
	}

	function Deactivate(main) {
		main.ListenOnSecond.Remove(self);
	}

	function OnSecond() {
		if (self._end) {
			return;
		}

		winner = null;
		aliveCount = 0;
		playerCount = 0;

		for (p in Network.Players) {
			if (p.GetCustomProperty("CurrentMode") != "BOMB") {
				continue;
			}

			if (p.Status != "Spectating") {
				playerCount += 1;
			}

			if (p.Status == "Alive") {
				winner = p;
				aliveCount += 1;
			}
		}

		if (playerCount < 2) {
			NetworkUtil.SetCustomProperty("BOMB_status", "Waiting for players.");
			return;
		}

		NetworkUtil.SetCustomProperty("BOMB_status", "Players remaining: " + aliveCount);

		if (aliveCount <= 1) {
			reason = "<b>Nobody</b> won.";
			if (aliveCount == 1) {
				reason = "<b>" + winner.Name + "</b> won.";
			}

			data = Dict();
			data.Set("op", "BOMB_end");
			data.Set("reason", reason);
			Network.SendMessageAll(Json.SaveToString(data));
			NetworkUtil.SetCustomProperty("BOMB_status", "Round ending.");

			self._end = true;
			self._bombRound += 1;
			__bombRound = self._bombRound;
			NetworkUtil.SetCustomProperty("bombround", __bombRound);
			self.Restart(winner);
		}
	}

	coroutine Restart(winner) {
		wait 5.23;

		data = Dict();
		data.Set("op", "BOMB_start");
		Network.SendMessageAll(Json.SaveToString(data));
		self._end = false;
		self._bombRound += 1;
	}
}

class AsoLogic {
	_lastRetry = -5;
	_startTime = 0;
	_prevTime = 0;

	function Init(map) {
		self._map = map;
		self._playerSpawner = PlayerSpawner(map);
	}

	function Activate(main) {
		main.ListenOnCharacterSpawn.Add(self);
		main.ListenOnTick.Add(self);
		main.ListenOnAction.Add(self);

		main.UX.AddBind("Interaction/Function1", "Retry");

		self.Retry();
	}

	function Deactivate(main) {
		main.ListenOnCharacterSpawn.Remove(self);
		main.ListenOnTick.Remove(self);
		main.ListenOnAction.Remove(self);
	}

	function OnAction(action) {
		if (action == "Retry") {
			self.Retry();
		}
	}

	function OnTick() {
		wave = Network.MasterClient.GetCustomProperty("ASO_wave");
		titanCount = Network.MasterClient.GetCustomProperty("ASO_titanCount");
		totalKills = Network.MyPlayer.Kills;
		totalDamage = Network.MyPlayer.TotalDamage;
		roundTime = Math.Round(self.GetTime());

		if (self._prevTime != roundTime) {
			self.OnSecondTransition(roundTime);
			self._prevTime = roundTime;
		}

		label = "Titans Left: " + titanCount + "  Wave: " + wave + String.Newline;
		label += "Time: " + roundTime + "s  Kills: " + totalKills + "  Damage: " + totalDamage;
		UI.SetLabel("TopCenter", label);
	}

	function Retry() {
		timeLeft = 5.0 - (Time.GameTime - self._lastRetry);
		if (Time.GameTime - self._lastRetry < 5.0) {
			Game.Print("<color=red>You must wait " + String.FormatFloat(timeLeft, 2) + "s before you can retry.");
			return;
		}

		self._lastRetry = Time.GameTime;

		data = Dict();
		data.Set("op", "ASO_restart");
		Network.SendMessage(Network.MasterClient, Json.SaveToString(data));

		self._playerSpawner.Spawn();
		self._startTime = Time.GameTime;
		Network.MyPlayer.ClearKDR();

		status = "[ASO] 0k in 0s";
		NetworkUtil.SetCustomProperty("ScoreStatus", status);
	}

	function OnSecondTransition(roundTime) {
		if (Math.Mod(roundTime, 15) == 0) {
			displayTime = Math.Ceil(self.GetTime());
			status = "[ASO] " + Math.Round(Network.MyPlayer.TotalDamage / 1000) + "k in " + displayTime + "s";
			NetworkUtil.SetCustomProperty("ScoreStatus", status);
		}

		if (roundTime == 100 || roundTime == 300 || roundTime == 600) {
			kDamage = Math.Round(Network.MyPlayer.TotalDamage / 1000.0);
			Game.PrintAll("[ASO] <color=white>" + Network.MyPlayer.Name + "</color> hit " + kDamage + "k in " + roundTime + "s.");
			self.SubmitScore(roundTime);
		} elif (roundTime == 602 && Network.MyPlayer.Character != null) {
			Network.MyPlayer.Character.GetKilled("");
		} elif (roundTime == 605) {
			self.Retry();
		}
	}

	function SubmitScore(roundTime) {
		data = Dict();
		data.Set("op", "PB_submit");
		data.Set("category", Convert.ToString(roundTime) + "s");
		data.Set("score", Network.MyPlayer.TotalDamage);
		Network.SendMessage(Network.MasterClient, Json.SaveToString(data));
	}

	function OnCharacterSpawn(character) {
		if (character.IsMine && character.Type == "Human") {
			character.SetWeapon("Blades");
			character.Team = "Human";
		}
	}

	function GetTime() {
		return Time.GameTime - self._startTime;
	}

	function CanJoin() {
		for (p in Network.Players) {
			if (p.GetCustomProperty("CurrentMode") == "ASO") {
				Game.Print("<color=white>" + p.Name + "</color> is already on ASO.");
				return false;
			}
		}

		return true;
	}
}

class MC_AsoLogic {
	StartTitans = 3;
	AddTitansPerWave = 1;

	_wave = 0;

	function Init(map) {
		self._titanSpawner = TitanSpawner(map);
		self.NextWave();
	}

	function Activate(main) {
		main.ListenOnNetworkMessage.Add(self);
		main.ListenOnSecond.Add(self);
		main.ListenOnCharacterDie.Add(self);
	}

	function Deactivate(main) {
		main.ListenOnNetworkMessage.Remove(self);
		main.ListenOnSecond.Remove(self);
		main.ListenOnCharacterDie.Remove(self);
	}

	function OnNetworkMessage(sender, message) {
		data = Json.LoadFromString(message);

		if (data.Get("op") == "ASO_restart") {
			self.Restart();
		}
	}

	function OnSecond() {
		self.UpdateStatus();
	}

	function OnCharacterDie(victim, killer, killerName) {
		self._titanSpawner.OnCharacterDie(victim, killer, killerName);

		if (killer != null && killer.Type == "Human" && self._titanSpawner.GetCount() == 0) {
			self.NextWave();
		}

		if (killer != null) {
			self.UpdateStatus();
		}
	}

	function Restart() {
		self._titanSpawner.Clear();
		self._wave = 0;
		self.NextWave();
	}

	function UpdateStatus() {
		NetworkUtil.SetCustomProperty("ASO_wave", self._wave);
		NetworkUtil.SetCustomProperty("ASO_titanCount", self._titanSpawner.GetCount());
	}

	function NextWave() {
		self._wave += 1;
		amount = self.StartTitans + (self._wave - 1) * self.AddTitansPerWave;
		self._titanSpawner.SpawnTitans(amount);
	}
}

class AkinaLogic {
	_playerSpawner = null;
	_mapLoaded = false;
	_startTime = 0;
	_startBarriers = null;
	_started = false;
	_finished = false;

	function Init(mapOffset) {
		self._mapOffset = mapOffset;
	}

	function Activate(main) {
		main.ListenOnTick.Add(self);
		main.ListenOnAction.Add(self);
		main.ListenOnLoadMap.Add(self);
		main.ListenOnFinishRace.Add(self);

		main.UX.AddBind("Interaction/Function1", "Retry");

		if (self._mapLoaded) {
			self.Reset();
		} else {
			main.MapLoader.Load("Akina");
		}

		NetworkUtil.SetCustomProperty("ScoreStatus", "[AKINA]");
	}

	function Deactivate(main) {
		main.ListenOnTick.Remove(self);
		main.ListenOnAction.Remove(self);
		main.ListenOnLoadMap.Remove(self);
		main.ListenOnFinishRace.Remove(self);

		main.UX.RemoveBind("Interaction/Function1", "Retry");
	}

	function OnAction(action) {
		if (action == "Retry") {
			self.Reset();
		}
	}

	function OnLoadMap(map) {
		map.Translate(self._mapOffset);

		self._playerSpawner = PlayerSpawner(map);
		self._startBarriers = map.FindMapObjectsByName("Racing Start Barrier Cuboid");

		self._mapLoaded = true;
		self.Reset();
	}

	function OnFinishRace() {
		if (self._finished) {
			return;
		}

		finishTime = String.FormatFloat(self.GetRaceTime(), 2);
		UI.SetLabelForTime("MiddleCenter", "<color=white>" + Network.MyPlayer.Name + "</color> finished in " + finishTime + "s!", 5.0);
		Game.PrintAll("<color=white>" + Network.MyPlayer.Name + "</color> finished Akina in " + finishTime + "s!");
		NetworkUtil.SetCustomProperty("ScoreStatus", "[AKINA]" + finishTime + "s");

		self.SubmitScore();
		self._finished = true;
	}

	function OnTick() {
		if (!self._mapLoaded) {
			return;
		}

		if (self.GetRaceTime() > 0) {
			if (!self._started) {
				for (barrier in self._startBarriers) {
					barrier.Active = false;
				}
				self._started = true;
			}

			UI.SetLabel("TopCenter", "Racing Time: " + String.FormatFloat(self.GetRaceTime(), 2));
		} else {
			UI.SetLabel("TopCenter", "Race Starts In: " + String.FormatFloat(Math.Abs(self.GetRaceTime()), 2));
		}
	}

	function GetRaceTime() {
		return Time.GameTime - self._startTime;
	}

	function Reset() {
		for (barrier in self._startBarriers) {
			barrier.Active = true;
		}

		self._playerSpawner.Spawn();
		self._started = false;
		self._finished = false;
		self._startTime = Time.GameTime + 5.0;
	}

	function SubmitScore(roundTime) {
		data = Dict();
		data.Set("op", "PB_submit");
		data.Set("category", "Akina");
		data.Set("score", Math.Round(self.GetRaceTime() * 100));
		Network.SendMessage(Network.MasterClient, Json.SaveToString(data));
	}

		function CanJoin() { return true; }
}

	function Activate(main) {
		main.ListenOnNetworkMessage.Add(self);
	}

	function Deactivate(main) {
		main.ListenOnNetworkMessage.Remove(self);
	}

	function OnNetworkMessage(sender, message) {
		data = Json.LoadFromString(message);

		if (data.Get("op") == "MAP_request") {
			name = data.Get("name");
			self.Transfer(sender, name);
		}
	}

	function Transfer(player, name) {
		if (!self._transfer.Contains(name)) {
			return;
		}

		self.TransferAsync(player, name);
	}

	function TransferAsync(player, name) {
		payload = self._transfer.Get(name);

		totalLength = String.Length(payload);
		chunkCount = Math.Ceil(totalLength / 10000.0);

		for (i in Range(0, chunkCount, 1)) {
			chunk = String.SubstringWithLength(payload, i * 10000, Math.Min(totalLength, 10000));

			data = Dict();
			data.Set("op", "MAP_chunk");
			data.Set("name", name);
			data.Set("body", chunk);
			Network.SendMessage(player, Json.SaveToString(data));

			totalLength -= 10000;
			wait 0.5;
		}

		data = Dict();
		data.Set("op", "MAP_done");
		data.Set("name", name);
		Network.SendMessage(player, Json.SaveToString(data));
	}

	function CreateTransfer(name) {
		fileName = "Map_" + name;

		if (!PersistentData.FileExists(fileName)) {
			Game.Print("<color=red>Missing map '" + name + "'.</color>");
			return;
		}

		map = MapUtil.LoadFromFile(fileName);
		self._transfer.Set(name, Json.SaveToString(map));
	}
}

class MapLoaderLogic {
	_downloads = Dict();

	function Activate(main) {
		main.ListenOnNetworkMessage.Add(self);
	}

	function Deactivate(main) {
		main.ListenOnNetworkMessage.Remove(self);
	}

	function OnNetworkMessage(sender, message) {
		if (sender != Network.MasterClient) {
			return;
		}

		data = Json.LoadFromString(message);

		if (data.Get("op") == "MAP_chunk") {
			self.SaveChunk(data);
		} elif (data.Get("op") == "MAP_done") {
			self.FinishDownload(data);
		}
	}

	function FinishDownload(data) {
		name = data.Get("name");
		map = Json.LoadFromString(self._downloads.Get(name));
		self._downloads.Remove(name):

		MapUtil.SaveToFile("Map_" + name, map);
		objects = self.CreateMap(map);

		MainController.OnLoadMap(MapSubset(objects));

		Game.Print("Finished downloading map '" + name + "'.");
	}

	function SaveChunk(data) {
		name = data.Get("name");
		chunk = data.Get("body");
		self._downloads.Set(name, self._downloads.Get(name, "") + chunk);
	}

	function Load(name) {
		fileName = "Map_" + name;
		if (PersistentData.FileExists(fileName)) {
			map = MapUtil.LoadFromFile(fileName);
			objects = self.CreateMap(map);
			MainController.OnLoadMap(MapSubset(objects));
			Game.Print("Loaded '" + name + "' from file.");
		} else {
			self.RequestMap(name);
		}
	}

	function CreateMap(map) {
		t = map.Get("t", "");
		N = map.Get("n", 0);
		M = String.Length(t);

		columns = List();

		for (i in Range(0, M, 1)) {
			b = String.SubstringWithLength(t, i, 1);
			data = map.Get(Convert.ToString(i), "");
			if (b == "0") {
				columns.Add(self.DecompressColumnRLE(data));
			} elif (b == "1") {
				columns.Add(self.DecompressColumn(data));
			}
		}

		objects = List();

		for (k in Range(N)) {
			objects.Add(self.LoadRow(columns, k, M));
		}

		return objects;
	}

	function RequestMap(name) {
		data = Dict();
		data.Set("op", "MAP_request");
		data.Set("name", name);
		Network.SendMessage(Network.MasterClient, Json.SaveToString(data));
	}

	function LoadRow(columns, k, M) {
		row = List();
		for (i in Range(M)) {
			row.Add(columns.Get(i).Get(k));
		}

		c = null;
		if (row.Get(22) != ";" && row.Get(22) != "") {
			c = String.Split(row.Get(22), "|").Get(0);
			row.Set(22, "");
		}

		row = String.Join(row, ",");
		obj = Map.CreateMapObjectRaw(row);

		if (c != null) {
			obj.AddComponent(c);
		}

		return obj;
	}

	function DecompressColumn(data) {
		lzw = LZW(data);
		return String.Split(lzw.Decompress(), ",");
	}

	function DecompressColumnRLE(data) {
		lzw = LZW(data);
		items = String.Split(lzw.Decompress(), ",");

		column = List();
		for (i in Range(0, items.Count, 2)) {
			a = Convert.ToInt(items.Get(i + 0));
			b = items.Get(i + 1);
			for (j in Range(0, a, 1)) {
				column.Add(b);
			}
		}

		return column;
	}
}

class CanvasLogic {
  _hold = false;
  _wasHold = false;
  _lines = Dict();
  _curveCache = Dict();
  _lineID = 0;
  
  function Init(map) {
    self.MapObject = map.FindMapObjectByName("Board");
    self._T = self.MapObject.QuaternionRotation * Vector3(1, 0, 0);
    self._B = self.MapObject.QuaternionRotation * Vector3(0, 0, 1);
    self._N = self.MapObject.QuaternionRotation * Vector3(0, 1, 0);
    self._base = self.MapObject.Position - (self._T * self.MapObject.Scale.X + self._B * self.MapObject.Scale.Z) * 0.5;
    self._pathCompressor = PathCompressor();
    self._segmentGenerator = SegmentGenerator(self.MapObject.Scale.Z * 0.005, self);
    self._preview = self.CreateLine(Color(0, 0, 0));
    self._bound = Vector2(self.MapObject.Scale.X, self.MapObject.Scale.Z);
    self._serializer = Serializer(self._bound);
  }

  function Activate(main) {
    main.ListenOnTick.Add(self);
    main.ListenOnFrame.Add(self);
    main.ListenOnChatInput.Add(self);
    main.ListenOnPlayerJoin.Add(self);
    main.ListenOnNetworkMessage.Add(self);
  }
  
  function Deactivate(main) {
    main.ListenOnTick.Remove(self);
    main.ListenOnFrame.Remove(self);
    main.ListenOnChatInput.Remove(self);
    main.ListenOnPlayerJoin.Remove(self);
    main.ListenOnNetworkMessage.Remove(self);
  }

  function OnTick() {
    mouse = Input.GetMouseAim() - self._base;
    X = Vector3.Dot(mouse, self._T);
    Y = Vector3.Dot(mouse, self._B);
    Z = Vector3.Dot(mouse, self._N);
    
    if (X < 0 || Y < 0 || X > self.MapObject.Scale.X || Y > self.MapObject.Scale.Z || Z < -0.1 || Z > 0.1) {
      return;
    }
    
    if (self._hold) {
      self._segmentGenerator.AddPoint(Vector2(mouse.X, mouse.Y));
      self._wasHold = true;
    } elif (self._wasHold) {
      self._lineID += 1;
      path = self._pathCompressor.Snip();

      if (path.GetCurves().Count > 0) {
        body = self._serializer.EncodePath(path);
        self.SendPath(null, Network.MyPlayer.ID + "_" + self._lineID, body);
      }
      
      self._preview.PositionCount = 0;
      self._segmentGenerator.Reset();
      
      self._wasHold = false;
    }
  }
  
  function OnPlayerJoin(player) {
    if (Network.IsMasterClient) {
      self.SyncCache(player);
    }
  }
  
  function OnFrame() {
    self._hold = Input.GetKeyHold("Human/AttackDefault");
    if (Input.GetKeyDown("Human/AttackSpecial")) {
      self.DeleteLine(self._lineID);
    }
  }
  
  function OnNetworkMessage(sender, message) {
    data = Json.LoadFromString(message);
    
    if (data.Get("op") == "CANVAS_draw") {
      id = data.Get("id");
      body = data.Get("body");
      path = self._serializer.DecodePath(body);
      line = self.DrawPath(path);
      self._lines.Set(id, line);
      
      if (Network.IsMasterClient) {
        self._curveCache.Set(id, body);
      }
    } elif (data.Get("op") == "CANVAS_delete") {
      id = data.Get("id");
      if (self._lines.Contains(id)) {
        self._lines.Get(id).Destroy();
        self._lines.Remove(id);
      }
      
      if (Network.IsMasterClient && self._curveCache.Contains(id)) {
        self._curveCache.Remove(id);
      }
    } elif (data.Get("op") == "CANVAS_clear" && sender == Network.MasterClient) {
      for (line in self._lines.Values) {
        line.Destroy();
      }
      self._lines.Clear();
    }
  }

  function OnChatInput(message) {
    if (Network.IsMasterClient && message == "/canvas_clear") {
      self.ClearCanvas();
      return false;
    }
  }
  
  function ClearCanvas() {
    self._curveCache.Clear();
    data = Dict();
    data.Set("op", "CANVAS_clear");
    Network.SendMessageAll(Json.SaveToString(data));
  }

  coroutine SyncCache(player) {
    for (key in self._curveCache.Keys) {
      self.SendPath(player, key, self._curveCache.Get(key));
      wait 0.25;
    }
  }
  
  function SendPath(player, id, body) {
    data = Dict();
    data.Set("op", "CANVAS_draw");
    data.Set("id", id);
    data.Set("body", body);
    
    if (player == null) {
      Network.SendMessageAll(Json.SaveToString(data));
    } else {
      Network.SendMessage(player, Json.SaveToString(data));
    }
  }
  
  function DeleteLine(id) {
    data = Dict();
    data.Set("op", "CANVAS_delete");
    data.Set("id", Network.MyPlayer.ID + "_" + self._lineID);
    Network.SendMessageAll(Json.SaveToString(data));
    self._lineID -= 1;
  }
  
  function DrawPath(path) {
    line = self.CreateLine(Color(0, 0, 0));
    for (curve in path.GetCurves()) {
      offset = line.PositionCount;
      line.PositionCount += 11;
      for (i in Range(0, 11, 1)) {
        t = i / 10.0;
        p = curve.Sample(t);
        line.SetPosition(offset + i, self.ToWorldSpace(p));
      }
    }
    return line;
  }
  
  function OnAddSegment(point) {
    self._pathCompressor.OnAddSegment(point);
    self._preview.PositionCount += 1;
    self._preview.SetPosition(self._preview.PositionCount - 1, self.ToWorldSpace(point));
  }
  
  function CreateLine(color) {
    line = LineRenderer.CreateLineRenderer();
    line.Enabled = true;
    line.LineColor = color;
    line.PositionCount = 0;
    line.StartWidth = 0.05;
    line.EndWidth = 0.05;
    return line;
  }
  
  function ToWorldSpace(p) {
    return self._base + p.X * self._T + p.Y * self._B - 0.1 * self._N;
  }
}

class Leaderboard {
	_scores = Dict();

	function Init(name, isAscending, displayType) {
		self._isAscending = isAscending;
		self._fileName = "Leaderboard_" + name;
		self._displayType = displayType;
		self.Load();
	}

	function Submit(name, score) {
		if (self._scores.Contains(name)) {
			if (self._isAscending) {
					if (score > self._scores.Get(name)) { return; }
			} else {
					if (score < self._scores.Get(name)) { return; }
			}
		}

		self._scores.Set(name, score);
		self.Save();
	}

	function Load() {
		if (!PersistentData.FileExists(self._fileName)) {
			return;
		}

		PersistentData.Clear();
		PersistentData.LoadFromFile(self._fileName, false);
		data = PersistentData.GetProperty("scores", "");

		self._scores = Json.LoadFromString(data);
	}

	function Save() {
		PersistentData.Clear();
		PersistentData.SetProperty("scores", Json.SaveToString(self._scores));
		PersistentData.SaveToFile(self._fileName, false);
	}

	function Retrieve() {
		result = List();

		index = 0;
		total = 0;
		names = List();
		scores = List();

		for (key in self._scores.Keys) {
			value = self._scores.Get(key);
			scores.Add(Convert.ToInt(value) * 256 + index);
			names.Add(key);
			index += 1;
		}

		scores.Sort();

		if (self._isAscending) {
			startIndex = 0;
			endIndex = scores.Count;
			step = 1;
		} else {
			startIndex = scores.Count - 1;
			endIndex = -1;
			step = -1;
		}

		result = List();
		index = 1;

		for (i in Range(startIndex, endIndex, step)) {
			score = Math.Floor(scores.Get(i) / 256);
			if (self._displayType == "k") {
				score = Math.Round(score / 1000.0) + "k";
			} elif (self._displayType == "s") {
				score = String.FormatFloat(score / 100.0, 2) + "s";
			} else {
				score = entry.Get(1);
			}

			name = names.Get(Math.Mod(scores.Get(i), 256));

			entry = List();
			entry.Add(name);
			entry.Add(score);
			result.Add(entry);

			index += 1;
		}

		return result;
	}
}

class LZW {
	function Init(data) {
		self._data = data;

		self._firstIndex = 0;
		self._secondIndex = 0;
		self._firstLetter = "0";

		self._dict = Dict();
		self._charset = "0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ !#$%&'()*+,-./:;<=>?@[\]^_`{|}~";
		self._charsetSize = String.Length(self._charset);
		self.InitDict();

		self._res = self._dict.Get(self.GetBlock(0));
		self._p = self._res;
	}

	function GetBlock(i) {
		return String.SubstringWithLength(self._data, i * 2, 2);
	}

	function GetResult() {
		return self._res;
	}

	function NextKey() {
		res = self._firstLetter + self.GetCharsetAt(self._secondIndex);

		self._secondIndex += 1;
		if (self._secondIndex == self._charsetSize) {
			self._secondIndex = 0;
			self._firstIndex += 1;
			self._firstLetter = self.GetCharsetAt(self._firstIndex);
		}

		return res;
	}

	function GetCharsetAt(i) {
		return String.SubstringWithLength(self._charset, i, 1);
	}

	function Decompress() {
		for (i in Range(1, String.Length(self._data) / 2, 1)) {
			c = self.GetBlock(i);
			if (!self._dict.Contains(c)) {
				newEntry = self._p + String.SubstringWithLength(self._p, 0, 1);
				self._dict.Set(self.NextKey(), newEntry);
				self._res += newEntry;
				self._p = newEntry;
			} else {
				code = self._dict.Get(c);
				self._res += code;
				newEntry = self._p + String.SubstringWithLength(code, 0, 1);
				self._dict.Set(self.NextKey(), newEntry);
				self._p = code;
			}
		}

		return self._res;
	}

	function InitDict() {
		for (i in Range(0, self._charsetSize, 1)) {
			c = self.GetCharsetAt(i);
			self._dict.Set(self.NextKey(), c);
		}
	}
}

class PathCompressor {
  MinSampleCount = 10;
  MaxSampleCount = 50;
  MinCorrelation = 0.95;
  MaxCosAngle = 0.8;
  
  _prevPoint = null;
  _prevSample = null;
  _pointBuffer = List();
  _sampleBuffer = List();
  _pearson = PearsonCorrelation();
  _curves = List();
  
  function FlushBuffer() {
    self._pearson.Reset();
    self._sampleBuffer.Clear();
    self._pointBuffer.Clear();
    self._prevSample = null;
  }
  
  function OnAddSegment(point) {
    if (self._prevPoint == null) {
      self._prevPoint = point;
    } else {
      delta = point - self._prevPoint;
      self.OnSample(delta);
      self._pointBuffer.Add(point);
      self._prevPoint = point;
    }
  }
  
  function Snip() {
    if (self._pointBuffer.Count > 2) {
      curve = CurveUtil.FitBezier(self._pointBuffer);
      self._curves.Add(curve);
    }
    
    curves = self._curves;
    self._curves = List();
    self.FlushBuffer();
    self._prevPoint = null;
    
    return BezierPath(curves);
  }
  
  function OnSample(sample) {
    self._pearson.AddSample(sample);
    
    cosAngle = 1.0;
    if (self._prevSample != null) {
      cosAngle = Vector2.Dot(sample.Normalized, self._prevSample.Normalized);
    }

    r = self._pearson.GetCorrelation();
    shouldSplit = Math.Abs(r) < self.MinCorrelation || cosAngle < self.MaxCosAngle || self._sampleBuffer.Count > self.MaxSampleCount;
    
    if (shouldSplit) {
      curve = CurveUtil.FitBezier(self._pointBuffer);
      self._curves.Add(curve);
      self.FlushBuffer();
    } else {
      self._prevSample = sample;
    }
  }
}

extension CurveUtil {
  _pearson = PearsonCorrelation();
  
  function FitBezier(points) {
    if (points.Count < 5) {
      A = points.Get(0);
      C = points.Get(points.Count - 1);
      B = (A + C) * 0.5;
      return BezierCurve(A, B, C);
    }
    
    self._pearson.Reset();
    
    n = Convert.ToFloat(points.Count - 1);
    for (i in Range(1, points.Count, 1)) {
      sample = (points.Get(i) - points.Get(i - 1)) * n;
      self._pearson.AddSample(sample);
    }

    sx = self._pearson.GetX();
    sy = self._pearson.GetY();
    sxx = self._pearson.GetXX();
    syy = self._pearson.GetYY();
    sxy = self._pearson.GetXY();
    cov = sxy / n - sx * sy / n / n;
    
    sigmay2 = Math.Max(syy / n - sy * sy / n / n, 0.001);
    b1 = cov / sigmay2;
    b0 = (sy - sx * b1) / n;

    sigmax = Math.Sqrt(Math.Max(sxx / n - sx * sx / n / n, 0.001));
    x0 = sx / n - 2 * sigmax;
    x1 = sx / n + 2 * sigmax;

    p0 = Vector2(x0, b0 + x0 * b1);
    p1 = Vector2(x1, b0 + x1 * b1);
    
    sample1 = points.Get(1) - points.Get(0);
    sample2 = points.Get(points.Count - 1) - points.Get(points.Count - 2);
    sampleDelta = sample2 - sample1;
    flipAnchor = sampleDelta.X < 0;
    
    if (flipAnchor) {
      tmp = p0;
      p0 = p1;
      p1 = tmp;
    }

    A = points.Get(0);
    C = points.Get(n);
    B = (p0 + A * 2.0) * 0.5;

    return BezierCurve(A, B, C);
  }
}

class BezierPath {
  function Init(curves) {
    self._curves = curves;
  }
  
  function GetCurves() {
    return self._curves;
  }
}

class BezierCurve {
  function Init(A, B, C) {
    self.A = A;
    self.B = B;
    self.C = C;
  }

  function Sample(t) {
    const a = (1.0 - t) * self.A + t * self.B;
    const b = (1.0 - t) * self.B + t * self.C;
    return (1.0 - t) * a + t * b;
  }
}

class PearsonCorrelation {
  function Init() {
    self.Reset();
  }

  function Reset() {
    self._sx = 0.0;
    self._sy = 0.0;
    self._sxx = 0.0;
    self._syy = 0.0;
    self._sxy = 0.0;
    self._sampleCount = 0;
  }

  function AddSample(sample) {
    self._sx += sample.X;
    self._sy += sample.Y;
    self._sxx += sample.X * sample.X;
    self._syy += sample.Y * sample.Y;
    self._sxy += sample.X * sample.Y;
    self._sampleCount += 1;
  }
  
  function GetX() { return self._sx; }
  function GetY() { return self._sy; }
  function GetXX() { return self._sxx; }
  function GetYY() { return self._syy; }
  function GetXY() { return self._sxy; }
  
  function GetSigmaX() {
    n = self._sampleCount;
    return Math.Sqrt(self._sxx / n - self._sx * self._sx / n / n);
  }
  
  function GetSigmaY() {
    n = self._sampleCount;
    return Math.Sqrt(self._syy / n - self._sy * self._sy / n / n);
  }
  
  function GetCorrelation() {
    n = self._sampleCount;
    cov = self._sxy / n - self._sx * self._sy / n / n;
    sigmax = self.GetSigmaX();
    sigmay = self.GetSigmaY();
    return cov / sigmax / sigmay;
  }
}

class SegmentGenerator {
  _prevPoint = null;
  _totalDist = 0.0;
  _segmentConsumer = null;

  function Init(segmentLength, segmentConsumer) {
    self._segmentLength = segmentLength;
    self._segmentConsumer = segmentConsumer;
  }

  function Reset() {
    self._prevPoint = null;
    self._totalDist = 0.0;
  }

  function AddPoint(point) {
    if (self._prevPoint == null) {
      self._prevPoint = point;
      return;
    }

    delta = point - self._prevPoint;
    dist = delta.Magnitude;

    tAccum = Math.Ceil(self._totalDist / self._segmentLength) * self._segmentLength;
    
    while (tAccum < self._totalDist + dist) {
      let t = (tAccum - self._totalDist) / dist;
      lerpPoint = self._prevPoint + (point - self._prevPoint) * t;
      self._segmentConsumer.OnAddSegment(lerpPoint);
      tAccum += self._segmentLength;
    }

    self._totalDist += dist;
    self._prevPoint = point;
  }
}

class Serializer {
  _count = 0;
  _charset = "0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ !#$%&'()*+,-./:;<=>?@[\]^_`{|}~";
  
  function Init(bound) {
    self._bound = bound;
    self._charsetDict = Dict();
    self._charsetSize = String.Length(self._charset);
    self._rangeBound = Convert.ToFloat(self._charsetSize * self._charsetSize);
    
    for (i in Range(0, self._charsetSize, 1)) {
      self._charsetDict.Set(String.SubstringWithLength(self._charset, i, 1), i);
    }
  }
  
  function DecodePath(message) {
    messageLength = String.Length(message);
    
    A = self.DecodeVector2(String.SubstringWithLength(message, 0, 4));
    
    curves = List();
    
    for (i in Range(4, messageLength, 8)) {
      B = self.DecodeVector2(String.SubstringWithLength(message, i, 4));
      C = self.DecodeVector2(String.SubstringWithLength(message, i + 4, 4));
      
      curves.Add(BezierCurve(A, B, C));
      
      A = C;
    }
    
    return BezierPath(curves);
  }
  
  function DecodeVector2(v) {
    a = self.DecodeRange(String.SubstringWithLength(v, 0, 2)) * self._bound.X;
    b = self.DecodeRange(String.SubstringWithLength(v, 2, 2)) * self._bound.Y;
    return Vector2(a, b);
  }
  
  function DecodeRange(v) {
    return self.DecodeInt(v) / self._rangeBound;
  }
  
  function DecodeInt(v) {
    x = String.SubstringWithLength(v, 0, 1);
    y = String.SubstringWithLength(v, 1, 1);
    a = self.CharsetTo(x);
    b = self.CharsetTo(y);
    return a * self._charsetSize + b;
  }
  
  function EncodePath(path) {
    result = "";
    curves = path.GetCurves();
    
    lastPoint = curves.Get(0).A;
    result += self.EncodeVector2(lastPoint);
    
    for (curve in curves) {
      result += self.EncodeVector2(curve.B);
      result += self.EncodeVector2(curve.C);
    }
    
    return result;
  }
  
  function EncodeVector2(v) {
    return self.EncodeRange(v.X / self._bound.X) + self.EncodeRange(v.Y / self._bound.Y);
  }
  
  function EncodeRange(v) {
    return self.EncodeInt(v * self._rangeBound);
  }
  
  function EncodeInt(v) {
    a = self.CharsetAt(Math.Mod(Math.Floor(v / self._charsetSize), self._charsetSize));
    b = self.CharsetAt(Math.Mod(v, self._charsetSize));
    return a + b;
  }
  
  function CharsetAt(i) {
    return String.SubstringWithLength(self._charset, i, 1);
  }
  
  function CharsetTo(c) {
    return self._charsetDict.Get(c);
  }
}
